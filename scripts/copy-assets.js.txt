const fs = require('fs');
const path = require('path');

const origem = path.join(__dirname, '..', 'src', 'database', 'schema.sql');
const destinoDir = path.join(__dirname, '..', 'dist', 'database');
const destino = path.join(destinoDir, 'schema.sql');

if (!fs.existsSync(destinoDir)) {
  fs.mkdirSync(destinoDir, { recursive: true });
}

fs.copyFileSync(origem, destino);
console.log('schema.sql copiado para dist/database/schema.sql');
